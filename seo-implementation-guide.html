robots.ts

app/robots.ts
import type { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: ['/private/', '/admin/', '/backend/'],
      },
      {
        userAgent: 'Googlebot',
        allow: '/',
      },
      {
        userAgent: 'Bingbot',
        allow: '/',
      },
    ],
    sitemap: 'https://www.philippinescasinos.com/sitemap.xml',
    host: 'https://www.philippinescasinos.com',
  }
}

Paano i-implement:

  3. I-save ang file at i-deploy ang iyong Next.js application

import type { MetadataRoute } from 'next'

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = 'https://www.philippinescasinos.com'
  const currentDate = new Date().toISOString()
  
  // Main pages
  const mainPages = [
    {
      url: baseUrl,

    },

    },
    {

      lastModified: currentDate,
      changeFrequency: 'weekly',
      priority: 0.9,
    },
    {
      url: `${baseUrl}/reviews`,

      changeFrequency: 'weekly',
      priority: 0.8,
    },
    {
      url: `${baseUrl}/bonuses`,
      lastModified: currentDate,
      changeFrequency: 'weekly',
      priority: 0.8,
    },
    {
      url: `${baseUrl}/sports-betting`,
      lastModified: currentDate,
      changeFrequency: 'weekly',
      priority: 0.7,

  ]
  
  // Casino review pages
  const casinoReviews = [
    'smgame-casino',

  ].map(casino => ({
    url: `${baseUrl}/reviews/${casino}`,
    lastModified: currentDate,

Implementation Steps

Step 1: Create the Files

Gumawa ng lahat ng files na nabanggit sa itaas gamit ang provided code templates. I-customize ang mga ito para sa iyong specific website information.

Step 2: Upload to Your Server

I-upload ang lahat ng files sa tamang directories sa iyong web server:

  • robots.txt at sitemap.xml ay dapat nasa root directory
  • Para sa Next.js, ilagay ang sitemap.ts at robots.ts sa app directory
  • Ilagay ang iba pang files sa public directory

Step 3: Verify Implementation

  1. I-check na accessible ang bawat file sa pamamagitan ng pag-visit sa URL nito directly
  2. Gamitin ang mga sumusunod na tools para i-validate ang iyong files:
    • Google Search Console - I-submit at i-test ang iyong sitemap
    • XML Sitemap Validator - I-validate ang iyong sitemap.xml
    • Robots.txt Tester - I-test ang iyong robots.txt

Step 4: Submit to Search Engines

  1. Gumawa ng accounts sa:
    • Google Search Console
    • Bing Webmaster Tools
    • Yandex Webmaster
  2. I-verify ang ownership ng iyong website gamit ang provided methods
  3. I-submit ang iyong sitemap URL sa bawat search engine:
    • Sa Google Search Console: Go to Sitemaps > Add a new sitemap
    • Sa Bing Webmaster Tools: Go to Configure My Site > Sitemaps
    • Sa Yandex Webmaster: Go to Indexing > Sitemap files

Step 5: Monitor and Update

  1. Regular na i-check ang search engine consoles para sa crawling errors
  2. I-update ang iyong sitemap.xml tuwing magdadagdag ka ng bagong pages
  3. Panatilihing current ang lastmod dates sa iyong sitemap
  4. I-monitor ang indexing status ng iyong website sa search engines
